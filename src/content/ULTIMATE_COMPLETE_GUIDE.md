# Base URL: https://open-api.guesty.com/v1

# Step 1: Get Access Token
curl -X POST 'https://open-api.guesty.com/oauth2/token' \
  --header 'Accept: application/json' \
  --header 'Content-Type: application/x-www-form-urlencoded' \
  --data-urlencode 'grant_type=client_credentials' \
  --data-urlencode 'scope=open-api' \
  --data-urlencode 'client_secret=YOUR_CLIENT_SECRET' \
  --data-urlencode 'client_id=YOUR_CLIENT_ID'

# Response:
{
  "token_type": "Bearer",
  "expires_in": 86400,
  "access_token": "YOUR_ACCESS_TOKEN",
  "scope": "open-api"
}

# Token expires in 24 hours (86400 seconds)
# Maximum 5 tokens per 24 hours per clientId
# Best Practice: Cache token and reuse until expires

const SUPABASE_SERVICE_KEY = Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")!;

const corsHeaders = {
  "Access-Control-Allow-Origin": "*",
  "Access-Control-Allow-Headers": "authorization, x-client-info, apikey, content-type",
  "Access-Control-Allow-Methods": "POST, GET, OPTIONS",
};

interface QuoteRequest {
  propertyId: string;
  unitId: string;
  checkIn: string;
  checkOut: string;
  adults: number;
  children: number;
  addons: string[];
}

serve(async (req) => {
  // Handle CORS preflight
  if (req.method === "OPTIONS") {
    return new Response(null, { headers: corsHeaders });
  }

  if (req.method !== "POST") {
    return new Response("Method not allowed", { status: 405, headers: corsHeaders });
  }

  try {
    const body: QuoteRequest = await req.json();
    const { propertyId, unitId, checkIn, checkOut, adults, children, addons } = body;

    // Validate inputs
    if (!propertyId || !unitId || !checkIn || !checkOut) {
      return Response.json(
        { error: "Missing required fields" },
        { status: 400, headers: corsHeaders }
      );
    }

    const start = new Date(checkIn);
    const end = new Date(checkOut);
    
    if (isNaN(start.getTime()) || isNaN(end.getTime())) {
      return Response.json(
        { error: "Invalid date format" },
        { status: 400, headers: corsHeaders }
      );
    }
    
    const nights = Math.round((end.getTime() - start.getTime()) / (1000 * 60 * 60 * 24));

    if (nights < 1 || nights > 365) {
      return Response.json(
        { error: "Invalid stay duration" },
        { status: 400, headers: corsHeaders }
      );
    }

    // Check availability - CRITICAL: Prevent double-booking
    const stayRange = `[${checkIn}T15:00:00Z,${checkOut}T11:00:00Z)`;
    const { data: conflicts } = await supabase
      .from("reservation_units")
      .select("id")
      .eq("unit_id", unitId)
      .overlaps("stay", stayRange);

    if (conflicts && conflicts.length > 0) {
      return Response.json(
        { error: "Unit not available for selected dates" },
        { status: 409, headers: corsHeaders }
      );
    }

    // Check pending reservations (held by other guests)
    const { data: pendingConflicts } = await supabase
      .from("pending_reservations")
      .select("id")
      .eq("unit_id", unitId)
      .eq("status", "pending")
      .gt("expires_at", new Date().toISOString())
      .or(`check_in.lte.${checkOut},check_out.gte.${checkIn}`);

    if (pendingConflicts && pendingConflicts.length > 0) {
      return Response.json(
        { error: "Unit temporarily held by another guest" },
        { status: 409, headers: corsHeaders }
      );
    }

    // Get unit pricing
    const { data: unit, error: unitError } = await supabase
      .from("units")
      .select("base_price, property_id, max_guests")
      .eq("id", unitId)
      .single();

    if (unitError || !unit) {
      return Response.json(
        { error: "Unit not found" },
        { status: 404, headers: corsHeaders }
      );
    }

    // Validate guest count
    if (adults + children > unit.max_guests) {
      return Response.json(
        { error: `Exceeds maximum occupancy of ${unit.max_guests} guests` },
        { status: 400, headers: corsHeaders }
      );
    }

    // Calculate pricing
    const basePrice = Number(unit.base_price);
    let adjustedSubtotal = basePrice * nights;

    // Weekend multiplier (Friday=5, Saturday=6)
    const { data: ratePlan } = await supabase
      .from("rate_plans")
      .select("weekend_multiplier")
      .eq("unit_id", unitId)
      .single();

    if (ratePlan) {
      let weekendNights = 0;
      for (let i = 0; i < nights; i++) {
        const currentDate = new Date(start);
        currentDate.setDate(start.getDate() + i);
        const day = currentDate.getDay();
        if (day === 5 || day === 6) weekendNights++;
      }
      const weekdayNights = nights - weekendNights;
      const weekendMultiplier = Number(ratePlan.weekend_multiplier);
      adjustedSubtotal = (weekdayNights * basePrice) + (weekendNights * basePrice * weekendMultiplier);
    }

    // Calculate fees
    const cleaningFee = Math.round(adjustedSubtotal * 0.12);
    const serviceFee = Math.round(adjustedSubtotal * 0.08);

    // Addons

    if (addons && addons.length > 0) {

        .from("addons")
        .select("title, price")
        .in("id", addons);
      if (addonData) {
        addonData.forEach((addon) => {
          addonTotal += Number(addon.price);
          addonLineItems.push({ label: addon.title, amount: Number(addon.price) });
        });
      }

      return Response.json({ error: "Quote not found or expired" }, { status: 404 });
    }

    // Get property
    const { data: property } = await supabase
      .from("properties")
      .select("name")
      .eq("id", pending.property_id)
      .single();

    // Create Stripe PaymentIntent
    const paymentIntent = await stripe.paymentIntents.create({
      amount: Math.round(Number(pending.total_amount) * 100),
      currency: pending.currency.toLowerCase(),
      automatic_payment_methods: { enabled: true },
      metadata: {
        quoteId: pending.id,
        propertyName: property?.name || "Property",
        checkIn: pending.check_in,
        checkOut: pending.check_out,
        guestEmail: guest.email,
      },
    });

    // Update pending reservation
    await supabase
      .from("pending_reservations")
      .update({
        guest_email: guest.email,
        guest_first_name: guest.firstName,
        guest_last_name: guest.lastName,
        guest_phone: guest.phone,
        stripe_payment_intent_id: paymentIntent.id,
      })
      .eq("id", quoteId);

    return Response.json({
      pendingId: pending.id,
      clientSecret: paymentIntent.client_secret,
    });

  } catch (err) {
    console.error("Create pending error:", err);
    return Response.json({ error: "Internal server error" }, { status: 500 });
  }
});

import { serve } from "https://deno.land/std@0.224.0/http/server.ts";
import { createClient } from "https://esm.sh/@supabase/supabase-js@2";
import Stripe from "https://esm.sh/stripe@14.11.0";

const SUPABASE_URL = Deno.env.get("SUPABASE_URL")!;
const SUPABASE_SERVICE_KEY = Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")!;
const STRIPE_SECRET_KEY = Deno.env.get("STRIPE_SECRET_KEY")!;
const STRIPE_WEBHOOK_SECRET = Deno.env.get("STRIPE_WEBHOOK_SECRET")!;

const supabase = createClient(SUPABASE_URL, SUPABASE_SERVICE_KEY);
const stripe = new Stripe(STRIPE_SECRET_KEY, {
  apiVersion: "2024-11-20.acacia",
  httpClient: Stripe.createFetchHttpClient(),
});

serve(async (req) => {
  const signature = req.headers.get("stripe-signature");
  if (!signature) {
    return new Response("No signature", { status: 400 });
  }

  try {
    const body = await req.text();
    const event = stripe.webhooks.constructEvent(body, signature, STRIPE_WEBHOOK_SECRET);

    if (event.type === "payment_intent.succeeded") {
      const paymentIntent = event.data.object;
      const quoteId = paymentIntent.metadata.quoteId;

      if (!quoteId) {
        return Response.json({ error: "No quoteId in metadata" }, { status: 400 });
      }

      // Fetch pending reservation
      const { data: pending } = await supabase
        .from("pending_reservations")
        .select("*")
        .eq("id", quoteId)
        .eq("status", "pending")
        .single();

      if (!pending) {
        return Response.json({ error: "Pending not found" }, { status: 404 });
      }

      // Create confirmed reservation
      const { data: reservation } = await supabase
        .from("reservations")
        .insert({
          property_id: pending.property_id,
          guest_email: pending.guest_email,
          guest_first_name: pending.guest_first_name,
          guest_last_name: pending.guest_last_name,
          guest_phone: pending.guest_phone,
          currency: pending.currency,
          total_amount: pending.total_amount,
          status: "confirmed",
          payment_status: "paid",
          stripe_payment_intent_id: paymentIntent.id,
          check_in: pending.check_in,
          check_out: pending.check_out,
        })
        .select()
        .single();

      if (!reservation) {
        return Response.json({ error: "Failed to create reservation" }, { status: 500 });
      }

      // Create reservation unit
      const stayRange = `[${pending.check_in}T15:00:00Z,${pending.check_out}T11:00:00Z)`;
      await supabase.from("reservation_units").insert({
        reservation_id: reservation.id,
        unit_id: pending.unit_id,
        stay: stayRange,
      });

      // Mark pending as converted
      await supabase
        .from("pending_reservations")
        .update({ status: "converted" })
        .eq("id", quoteId);

      console.log(`Reservation ${reservation.id} confirmed`);
    }

    return new Response("OK", { status: 200 });
  } catch (err) {
    console.error("Webhook error:", err);
    return new Response("Webhook error", { status: 400 });
  }
});

const SUPABASE_URL = Deno.env.get("SUPABASE_URL")!;
const SUPABASE_SERVICE_KEY = Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")!;
const GUESTY_API_KEY = Deno.env.get("GUESTY_API_KEY")!;
const GUESTY_ACCOUNT_ID = Deno.env.get("GUESTY_ACCOUNT_ID")!;

const supabase = createClient(SUPABASE_URL, SUPABASE_SERVICE_KEY);

serve(async (req) => {
  if (req.method !== "POST") {
    return new Response("Method not allowed", { status: 405 });
  }

  try {
    // Fetch listings from Guesty
    const response = await fetch(
      `https://open-api.guesty.com/v1/listings?accountId=${GUESTY_ACCOUNT_ID}`,
      {
        headers: {
          "Authorization": `Bearer ${GUESTY_API_KEY}`,
          "Accept": "application/json",
        },
      }
    );

    const data = await response.json();
    const listings = data.results || [];

    for (const listing of listings) {
      // Upsert property
      await supabase.from("properties").upsert({
        guesty_listing_id: listing._id,
        name: listing.title,
        slug: listing.nickname?.toLowerCase().replace(/\s+/g, '-') || listing._id,
        destination: listing.address?.city || '',
        description: listing.description || '',
        hero_image: listing.pictures?.[0]?.original || '',
        gallery: listing.pictures?.map((p: any) => p.original) || [],
        amenities: listing.amenities || [],
        max_guests: listing.accommodates || 2,
        rating: listing.rating?.avg || null,
      }, { onConflict: 'guesty_listing_id' });

      // Create units from listing
      const unitId = `${listing._id}-default`;
      await supabase.from("units").upsert({
        id: unitId,
        property_id: listing._id,  // Will be updated after property insert
        name: listing.nickname || 'Default Unit',
        max_guests: listing.accommodates || 2,
        base_price: listing.prices?.basePrice || 100,
        guesty_unit_id: listing._id,
      }, { onConflict: 'guesty_unit_id' });
    }

    return Response.json({ success: true, synced: listings.length });
  } catch (err) {
    console.error("Sync error:", err);
    return Response.json({ error: "Sync failed" }, { status: 500 });
  }
});

import { useStripe, useElements, PaymentElement } from '@stripe/react-stripe-js';
import { Button } from '@/components/ui/button';
import { useState } from 'react';

interface PaymentFormProps {
  onSuccess: () => void;
}

export function PaymentForm({ onSuccess }: PaymentFormProps) {
  const stripe = useStripe();
  const elements = useElements();
  const [isProcessing, setIsProcessing] = useState(false);
  const [errorMessage, setErrorMessage] = useState<string>();

  async function handleSubmit(e: React.FormEvent) {
    e.preventDefault();
    if (!stripe || !elements) return;

    setIsProcessing(true);
    setErrorMessage(undefined);

    const { error } = await stripe.confirmPayment({
      elements,
      confirmParams: {
        return_url: `${window.location.origin}/confirmation`,
      },
    });

    if (error) {
      setErrorMessage(error.message);
      setIsProcessing(false);
    } else {
      onSuccess();
    }
  }

  return (
    <form onSubmit={handleSubmit} className="space-y-4">
      <PaymentElement />
      {errorMessage && (
        <p className="text-red-500 text-sm">{errorMessage}</p>
      )}
      <Button
        type="submit"
        className="w-full"
        disabled={!stripe || isProcessing}
      >
        {isProcessing ? 'Processing...' : 'Pay Now'}
      </Button>
    </form>
  );
}

import { serve } from "https://deno.land/std@0.224.0/http/server.ts";
import { createClient } from "https://esm.sh/@supabase/supabase-js@2";

const supabase = createClient(Deno.env.get("SUPABASE_URL")!, Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")!);

const corsHeaders = {
  "Access-Control-Allow-Origin": "*",
  "Access-Control-Allow-Headers": "authorization, x-client-info, apikey, content-type",
};

serve(async (req) => {
  if (req.method === "OPTIONS") {
    return new Response(null, { headers: corsHeaders });
  }

  const url = new URL(req.url);
  const unitId = url.searchParams.get("unitId");
  const start = url.searchParams.get("start");
  const end = url.searchParams.get("end");

  if (!unitId || !start || !end) {
    return Response.json({ error: "Missing parameters" }, { status: 400, headers: corsHeaders });
  }

  // Get confirmed reservations
  const { data: reservations } = await supabase
    .from("reservation_units")
    .select("stay")
    .eq("unit_id", unitId);

  // Get calendar blocks
  const { data: blocks } = await supabase
    .from("calendar_blocks")
    .select("*")
    .eq("unit_id", unitId)
    .gte("block_date", start)
    .lte("block_date", end);

  // Get pending reservations
  const { data: pending } = await supabase
    .from("pending_reservations")
    .select("check_in, check_out")
    .eq("unit_id", unitId)
    .eq("status", "pending")
    .gt("expires_at", new Date().toISOString());

  // Generate calendar dates
  const dates = [];
  const current = new Date(start);
  const endDate = new Date(end);

  while (current <= endDate) {
    const dateStr = current.toISOString().split("T")[0];
    const dayOfWeek = current.getDay();
    
    let status = "available";
    let price = null;

    // Check blocks
    if (blocks?.some(b => b.block_date === dateStr)) {
      status = "blocked";
    }

    // Check reservations
    if (reservations?.some(r => {
      const [start, end] = r.stay.replace(/[\[\]]/g, "").split(",");
      return dateStr >= start.split("T")[0] && dateStr < end.split("T")[0];
    })) {
      status = "booked";
    }

    // Check pending
    if (pending?.some(p => dateStr >= p.check_in && dateStr < p.check_out)) {
      status = "pending";
    }

    // Weekend pricing
    if (dayOfWeek === 5 || dayOfWeek === 6) {
      // Would fetch from rate plan
    }

    dates.push({
      date: dateStr,
      status,
      price,
      available: status === "available"
    });

    current.setDate(current.getDate() + 1);
  }

  return Response.json({ dates }, { headers: corsHeaders });
});

export class APIError extends Error {
  constructor(
    message: string,
    public status: number,
    public code?: string
  ) {
    super(message);
    this.name = "APIError";
  }
}

export async function handleAPIError(error: unknown): Promise<string> {
  if (error instanceof APIError) {
    switch (error.status) {
      case 400:
        return `Invalid request: ${error.message}`;
      case 401:
        return "Please log in to continue";
      case 403:
        return "You don't have permission for this action";
      case 404:
        return "The requested resource was not found";
      case 409:
        return error.message; // Usually availability conflict
      case 429:
        return "Too many requests. Please wait a moment.";
      case 500:
        return "Server error. Please try again later.";
      default:
        return error.message;
    }
  }
  
  if (error instanceof Error) {
    return error.message;
  }
  
  return "An unexpected error occurred";
}

# src/components/ErrorBoundary.tsx
import { Component, ReactNode } from 'react';

interface Props {
  children: ReactNode;
  fallback?: ReactNode;
}

interface State {
  hasError: boolean;
  error?: Error;
}

export class ErrorBoundary extends Component<Props, State> {
  constructor(props: Props) {
    super(props);
    this.state = { hasError: false };
  }

  static getDerivedStateFromError(error: Error): State {
    return { hasError: true, error };
  }

  render() {
    if (this.state.hasError) {
      return this.props.fallback || (
        <div className="p-4 bg-red-50 text-red-600 rounded">
          <h2 className="font-bold">Something went wrong</h2>
          <p>{this.state.error?.message}</p>
          <button 
            onClick={() => this.setState({ hasError: false })}
            className="mt-2 text-sm underline"
          >
            Try again
          </button>
        </div>
      );
    }
    return this.props.children;
  }
}

# === IMPLEMENT RATE LIMIT HANDLER ===
# src/lib/rate-limiter.ts

interface RateLimitConfig {
  maxRequests: number;
  windowMs: number;
}

class RateLimiter {
  private requests: Map<string, number[]> = new Map();

  constructor(private config: RateLimitConfig) {}

  isAllowed(key: string): boolean {
    const now = Date.now();
    const windowStart = now - this.config.windowMs;
    
    const userRequests = this.requests.get(key) || [];
    const recentRequests = userRequests.filter(t => t > windowStart);
    
    if (recentRequests.length >= this.config.maxRequests) {
      return false;
    }
    
    recentRequests.push(now);
    this.requests.set(key, recentRequests);
    return true;
  }

  getRetryAfter(key: string): number {
    const requests = this.requests.get(key) || [];
    if (requests.length === 0) return 0;
    
    const oldest = Math.min(...requests);
    return Math.ceil((oldest + this.config.windowMs - Date.now()) / 1000);
  }
}

export const guestyRateLimiter = new RateLimiter({
  maxRequests: 4,  // Keep under 5 limit
  windowMs: 1000,
});

# === EXPONENTIAL BACKOFF ===
async function fetchWithRetry<T>(
  fn: () => Promise<T>,
  maxRetries = 3
): Promise<T> {
  for (let i = 0; i < maxRetries; i++) {
    try {
      return await fn();
    } catch (error: any) {
      if (error.status === 429 && i < maxRetries - 1) {
        const delay = Math.pow(2, i) * 1000;
        await new Promise(r => setTimeout(r, delay));
        continue;
      }
      throw error;
    }
  }
  throw new Error("Max retries exceeded");
}

create table pricing_rules (
  id uuid primary key default uuid_generate_v4(),
  property_id uuid references properties(id),
  name text not null,
  condition jsonb not null,  -- { "field": "date", "op": "weekend" }

  priority int default 0,
  active boolean default true,
  created_at timestamptz default now()
);

  Rule 2: High season (+50%)

  Rule 3: Last minute discount (-10%)

  Rule 4: Long stay discount (-15%)

    adjustment: { "type": "percentage", "value": -15 }
  
  Rule 5: Weekday discount (-5%)
    condition: { "field": "dayOfWeek", "in": [1, 2, 3, 4] }

*/

# === APPLY PRICING RULES ===
async function calculateDynamicPrice(
  basePrice: number,
  checkIn: Date,
  checkOut: Date,
  propertyId: string
): Promise<number> {
  const nights = Math.ceil((checkOut.getTime() - checkIn.getTime()) / (1000 * 60 * 60 * 24));
  
  const { data: rules } = await supabase
    .from("pricing_rules")
    .select("*")
    .eq("property_id", propertyId)
    .eq("active", true)
    .order("priority", { ascending: false });

  let price = basePrice * nights;
  const checkInDate = checkIn.getDate();

  for (const rule of rules || []) {
    const condition = rule.condition;
    let applies = false;

    if (condition.field === "dayOfWeek") {
      const day = checkIn.getDay();
      if (condition.in?.includes(day)) applies = true;
    }

    if (condition.field === "month") {
      const month = checkIn.getMonth() + 1;
      if (condition.in?.includes(month)) applies = true;
    }

    if (condition.field === "nights") {
      if (condition.gte && nights >= condition.gte) applies = true;
      if (condition.lte && nights <= condition.lte) applies = true;
    }

    if (applies) {
      const adj = rule.adjustment;
      if (adj.type === "percentage") {
        price *= (1 + adj.value / 100);
      } else if (adj.type === "fixed") {
        price += adj.value;
      }
    }
  }

  return Math.round(price);
}
